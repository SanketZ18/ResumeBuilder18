# Use the official Tomcat image with Java 17
FROM tomcat:9-jdk17

# Remove default apps
RUN rm -rf /usr/local/tomcat/webapps/*

# Copy your WAR file and rename it to ROOT.war so it runs at "/"
COPY ResumeBuilder.war /usr/local/tomcat/webapps/ROOT.war

# Expose port (not required for Render, but good practice)
EXPOSE 8080
